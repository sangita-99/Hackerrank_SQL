-- Sum of latitude values in range
SELECT ROUND(SUM(LAT_N), 4)
FROM STATION
WHERE LAT_N > 38.7880 
  AND LAT_N < 137.2345;

-- Maximum latitude below threshold
SELECT MAX(LAT_N)
FROM STATION
WHERE LAT_N < 137.2345;

-- Minimum latitude above threshold
SELECT MIN(LAT_N)
FROM STATION
WHERE LAT_N > 38.7780;
