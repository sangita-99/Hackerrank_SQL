-- Count duplicate city names
SELECT 
COUNT(CITY) - COUNT(DISTINCT CITY)
FROM STATION;

-- City with the shortest name
SELECT CITY, LENGTH(CITY)
FROM STATION
ORDER BY LENGTH(CITY) ASC, CITY
LIMIT 1;

-- City with the longest name
SELECT CITY, LENGTH(CITY)
FROM STATION
ORDER BY LENGTH(CITY) DESC, CITY
LIMIT 1;
